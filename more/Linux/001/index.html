    Linux命令大全

    本文整理了自己在日常学习科研中频繁使用的Linux命令，包括文件操作、系统监控与进程管理等内容，按功能分类整理，便于查阅与学习。

    # 📁 文件与目录操作

    ls -l               # 以列表形式显示文件，也可写成ll
ls -a               # 列出目录下所有文件，包括以.开头的隐藏文件
cd -                # 返回最近访问目录
pwd                 # 显示当前工作目录的绝对路径
mkdir folder        # 创建目录
rm -r folder        # 递归删除文件夹
touch file          # 创建文件
rm file             # 删除文件
cp a ../file        # 拷贝文件
mv old new          # 重命名或移动文件
which python        # 查找可执行文件路径 
shopt -s globstar   # 打开globstar功能
shopt -u globstar   # 关闭globstar功能

    # 🔍 文件内容查看

    cat file.txt       # 直接输出文件内容
cat a b > c        # 合并a和b，写入新文件
cat -n file        # 输出文件内容时加行号
more file.txt      # 分页查看，空格下一页，q退出
less file.txt      # 与more类似，/关键字：向下搜索，n/N：下/上一个匹配项
head -n 11 file    # 查看前11行，默认10行
tail -n 11 file    # 查看后11行，默认10行
tail -f file.log   # 循环读取，实时查看日志更新

    # 📦 文件压缩与解压

    tar -czvf a.tar.gz folder/     # 打包并压缩
tar -xzvf a.tar.gz             # 解压
unzip a.zip                    # 解压 zip 文件
gzip file                      # 压缩文件为 file.gz
gunzip file.gz                 # 解压 file.gz

    # 🧠 系统资源与进程管理

    uname -r          # 查阅Linux核心版本
top               # 实时查看系统资源占用
cpuinfo           # 查询当前服务器 CPU 的详细硬件信息
lscpu             # 快速查看核心和线程信息
cat /proc/cpuinfo # 原始详细数据（每个逻辑CPU的信息）
pbsnodes          # 查看所有节点的简要状态
pbsnodes -l       # 列出当前不可用节点（如 down/offline）
pbsnodes -l free  # 仅列出空闲节点
htop              # 更友好的 top（需安装）
ps aux            # 查看所有进程
kill PID          # 终止进程
df -h             # 查看磁盘空间
quota -s          # 查看当前文件系统的配额和用量
du -sh folder     # 查看文件夹大小
